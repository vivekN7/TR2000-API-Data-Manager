BEGIN
    INSERT INTO STG_OPERATORS (RAW_JSON_ID, ETL_RUN_ID, OPERATOR_ID, OPERATOR_NAME)
    SELECT 
        r.JSON_ID,
        r.ETL_RUN_ID,
        TO_NUMBER(jt.operator_id),
        jt.operator_name

        r.JSON_DATA,
        '$[*]'

            operator_name VARCHAR2(200) PATH '$.OperatorName'

    ) jt
    WHERE r.ENDPOINT_NAME = 'operators'

END;
